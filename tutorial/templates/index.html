from animaid import HTMLDict, HTMLList

d = HTMLDict({
    "Fruits": HTMLList(["Apple", "Banana", "Cherry"]).horizontal.gap("8px"),
    "Vegetables": HTMLList(["Carrot", "Broccoli", "Spinach"]).horizontal.gap("8px"),
    "Grains": HTMLList(["Rice", "Wheat", "Oats"]).horizontal.gap("8px"),
}).key_bold.key_color("#2563eb")

html = d.render()

from animaid import HTMLDict, HTMLList

cards = [
    HTMLDict({"Name": "Alice", "Role": "Developer"}).key_bold.padding("12px").border("1px solid #e5e7eb").border_radius("8px"),
    HTMLDict({"Name": "Bob", "Role": "Designer"}).key_bold.padding("12px").border("1px solid #e5e7eb").border_radius("8px"),
    HTMLDict({"Name": "Carol", "Role": "Manager"}).key_bold.padding("12px").border("1px solid #e5e7eb").border_radius("8px"),
]

lst = HTMLList(cards).horizontal.gap("16px")

html = lst.render()
